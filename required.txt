
# Required for build

    export ANDROID_NDK=/opt/android/sdk/ndk/21.1.6352462

# Android build

    mkdir -p build
    cd build
    cmake -DVS_PLATFORM="android" -DVS_CUSTOMER="Virgil" ..
    make
    make apk

    mkdir -p build
    cd build
